# Frontend Integration

ConfidPay uses `@cofhe/sdk/web` for client-side encryption and contract interaction.

<Warning>
**Note:** `@cofhe/react` is in development. Use `@cofhe/sdk/web` directly for frontend integration.
</Warning>

## Installation

```bash
npm install @cofhe/sdk
npm install viem
```

## Core SDK Usage

### FhenixClient

The main client for encryption and decryption:

```typescript
import { FhenixClient } from '@cofhe/sdk/web';

const client = new FhenixClient();

// Encrypt data client-side
const encryptedSalary = await client.encrypt(5000000);
const encryptedRelease = await client.encrypt(
  Math.floor(Date.now() / 1000) + 30 * 24 * 60 * 60
);
const encryptedVestingRate = await client.encrypt(100);
```

### Encrypted Types (InEuint128)

Pass encrypted values to contracts using the `InEuint128` format:

```typescript
import { InEuint128 } from '@cofhe/sdk';

interface InEuint128 {
  publicAmount: bigint | string;
  // For re-encrypted values from FHE coprocessor
  encrypted?: EncryptedNumber;
}

// Usage with contract calls
await contract.createPayroll(
  recipient,
  { publicAmount: encryptedSalary },  // InEuint128 format
  { publicAmount: encryptedRelease },
  { publicAmount: encryptedVestingRate }
);
```

## React Integration

### Wrapper Component

Create a wrapper to manage the FhenixClient:

```typescript
// lib/fhenix.ts
import { FhenixClient } from '@cofhe/sdk/web';

let client: FhenixClient | null = null;

export function getFhenixClient(): FhenixClient {
  if (!client) {
    client = new FhenixClient();
  }
  return client;
}

export function useFhenix() {
  return getFhenixClient();
}
```

### Hook Pattern

```typescript
// hooks/useEncryption.ts
'use client';
import { useState, useCallback } from 'react';
import { getFhenixClient } from '@/lib/fhenix';

export function useEncryption() {
  const [isEncrypting, setIsEncrypting] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const encrypt = useCallback(async (value: number | bigint) => {
    setIsEncrypting(true);
    setError(null);
    try {
      const client = getFhenixClient();
      const encrypted = await client.encrypt(value);
      return encrypted;
    } catch (e) {
      setError(e as Error);
      throw e;
    } finally {
      setIsEncrypting(false);
    }
  }, []);

  return { encrypt, isEncrypting, error };
}
```

### Contract Write Hook

```typescript
// hooks/useContract.ts
'use client';
import { useState, useCallback } from 'react';
import { useWalletClient, usePublicClient, useAccount } from 'wagmi';
import { getFhenixClient } from '@/lib/fhenix';
import { ConfidPayABI } from '@/contracts/abi';

const CONFIDPAY_ADDRESS = process.env.NEXT_PUBLIC_CONFIDPAY_ADDRESS!;

export function useContractWrite() {
  const { address } = useAccount();
  const publicClient = usePublicClient();
  const { data: walletClient } = useWalletClient();
  const [isPending, setIsPending] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const write = useCallback(
    async (functionName: string, args: any[]) => {
      if (!walletClient || !address) throw new Error('Wallet not connected');
      
      setIsPending(true);
      setError(null);

      try {
        const hash = await walletClient.writeContract({
          address: CONFIDPAY_ADDRESS as `0x${string}`,
          abi: ConfidPayABI,
          functionName,
          args,
        });

        const receipt = await publicClient.waitForTransactionReceipt({ hash });
        return receipt;
      } catch (e) {
        setError(e as Error);
        throw e;
      } finally {
        setIsPending(false);
      }
    },
    [walletClient, publicClient, address]
  );

  return { write, isPending, error };
}
```

### Full Payroll Flow

```typescript
// components/CreatePayroll.tsx
'use client';
import { useState } from 'react';
import { useEncryption } from '@/hooks/useEncryption';
import { useContractWrite } from '@/hooks/useContract';

export function CreatePayroll() {
  const { encrypt, isEncrypting } = useEncryption();
  const { write, isPending } = useContractWrite();

  const createPayroll = async () => {
    const encryptedAmount = await encrypt(5000000); // 5 USDC
    const encryptedVestingRate = await encrypt(100);
    const encryptedNextRelease = await encrypt(
      Math.floor(Date.now() / 1000) + 30 * 24 * 60 * 60
    );

    await write('createPayroll', [
      employeeAddress,
      { publicAmount: encryptedAmount },
      { publicAmount: encryptedNextRelease },
      { publicAmount: encryptedVestingRate },
    ]);
  };

  return (
    <button onClick={createPayroll} disabled={isPending || isEncrypting}>
      {isPending ? 'Confirming...' : 'Create Payroll'}
    </button>
  );
}
```

## Error Handling

```typescript
const { encrypt, error: encryptError } = useEncryption();
const { write, error: writeError } = useContractWrite();

if (encryptError) {
  console.error('Encryption failed:', encryptError.message);
}

if (writeError) {
  console.error('Transaction failed:', writeError.message);
}
```

## Best Practices

1. **Always encrypt client-side** — Never send plaintext sensitive data
2. **Use InEuint128 format** — Pass `{ publicAmount: encrypted }` to contracts
3. **Handle loading states** — Show spinners during encryption/transactions
4. **Manage errors gracefully** — Provide user-friendly error messages
5. **Use appropriate types** — Convert amounts to correct decimal places
6. **Test on testnets** — Verify integration before mainnet deployment

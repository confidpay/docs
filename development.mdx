# Development

Set up a local development environment for building with ConfidPay.

## Prerequisites

| Requirement | Version | Notes |
|-------------|---------|-------|
| Node.js | 19+ | Required for modern JS features |
| npm/yarn/pnpm | Latest | Package manager |
| Git | Any | Version control |

## Installation

### 1. Clone the Starter

```bash
npx degit confidpay/cofhe-hardhat-starter my-confidpay-project
cd my-confidpay-project
npm install
```

### 2. Install Dependencies

```bash
npm install @fhenixprotocol/contracts @fhenixprotocol/cofhe-contracts
npm install @openzeppelin/contracts @reineiralabs/sdk
npm install @cofhe/react viem wagmi
```

  solidity: {
    version: '0.8.20',
    settings: {
      optimizer: {
        enabled: true,
        runs: 200,

  },

        url: 'https://sepolia.base.org',
        enabled: true,

## Deployment

### Deploy to Testnet

```bash
# Base Sepolia
npx hardhat run scripts/deploy.ts --network baseSepolia

# Arbitrum Sepolia
npx hardhat run scripts/deploy.ts --network arbitrumSepolia
```

### Deploy Script Example

```typescript
// scripts/deploy.ts
import { ethers } from 'hardhat';

async function main() {
  console.log('Deploying ConfidPay...');

  const ConfidPay = await ethers.getContractFactory('ConfidPay');
  const contract = await ConfidPay.deploy();

  await contract.waitForDeployment();
  const address = await contract.getAddress();

  console.log(`ConfidPay deployed to: ${address}`);
}

main().catch((error) => {
  console.error(error);
  process.exit(1);
});
```

## Testing

### Run Tests

```bash
# All tests
npx hardhat test

# Specific test file
npx hardhat test test/ConfidPay.test.ts

# With gas report
REPORT_GAS=true npx hardhat test
```

### Test Example

```typescript
// test/ConfidPay.test.ts
import { expect } from 'chai';
import { ethers } from 'hardhat';
import { FhenixHelpers } from '@fhenixprotocol/contracts';

describe('ConfidPay', function () {
  let confipay: any;
  let owner: any;
  let employee: any;

  beforeEach(async function () {
    [owner, employee] = await ethers.getSigners();

    const ConfidPay = await ethers.getContractFactory('ConfidPay');
    confipay = await ConfidPay.deploy();
    await confipay.waitForDeployment();
  });

  describe('Payroll Creation', function () {
    it('should create encrypted payroll', async function () {
      const encryptedAmount = FhenixHelpers.encrypt(5000000);
      const encryptedNextRelease = FhenixHelpers.encrypt(
        Math.floor(Date.now() / 1000) + 30 * 24 * 60 * 60
      );
      const encryptedVestingRate = FhenixHelpers.encrypt(100);

      await confipay.createPayroll(
        employee.address,
        encryptedAmount,
        encryptedNextRelease,
        encryptedVestingRate,
        0
      );

      const payroll = await confipay.payrolls(employee.address);
      expect(payroll.isActive).to.be.true;
    });

    it('should set correct ACL permissions', async function () {
      // Test ACL setup
      const canContractAccess = await confipay.canContractAccess(employee.address);
      expect(canContractAccess).to.be.true;
    });
  });
});
```

## Local Mock Testing

The Hardhat plugin provides a local CoFHE mock for testing:

```typescript
// hardhat.config.ts
import '@fhenixprotocol/hardhat-plugin';

const config: HardhatUserConfig = {
  // ... other config
  fhenix: {
    enableMock: true, // Use local mock instead of testnet
  },
};
```

Run tests without testnet dependency:

```bash
npx hardhat test --network hardhat
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Error: Cannot find module '@fhenixprotocol/contracts'">
    Run `npm install @fhenixprotocol/contracts @fhenixprotocol/cofhe-contracts`
  </Accordion>
  <Accordion title="Error: Invalid private key">
    Ensure your .env file has the correct format: `PRIVATE_KEY=0x...`
  </Accordion>
  <Accordion title="Error: Insufficient funds">
    Get testnet ETH from a faucet for Base Sepolia or Arbitrum Sepolia
  </Accordion>
  <Accordion title="Encryption/Decryption fails">
    Ensure ACL is set correctly with `FHE.allow()`, `FHE.allowThis()`, `FHE.allowSender()`
  </Accordion>
</AccordionGroup>

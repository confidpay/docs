# Development

## Installation

### Clone and Install

```bash
git clone https://github.com/confidpay/ConfidPay-Starter
cd ConfidPay-Starter/cofhe-hardhat-starter
npm install
```

### Install Dependencies

```bash
npm install @fhenixprotocol/cofhe-contracts
npm install @openzeppelin/contracts
npm install viem wagmi
npm install @cofhe/sdk
```

## Hardhat Configuration

```typescript
// hardhat.config.ts
import { HardhatUserConfig } from 'hardhat/config';
import '@nomicfoundation/hardhat-ethers';
import '@cofhe/hardhat-plugin';

const config: HardhatUserConfig = {
  solidity: {
    version: '0.8.25',
    settings: {
      optimizer: {
        enabled: true,
        runs: 200,
      },
      viaIR: true,  // Required for stack-too-deep issues
    },
  },
  networks: {
    hardhat: {
      forking: {
        url: 'https://sepolia.arbitrum.io/rpc',
        enabled: true,
      },
    },
    arbitrumSepolia: {
      url: 'https://sepolia.arbitrum.io/rpc',
      accounts: [process.env.PRIVATE_KEY!],
      chainId: 421614,
    },
  },
};

export default config;
```

## Deployment

### Deploy to Testnet

```bash
npx hardhat run scripts/deploy.ts --network arbitrumSepolia
```

### Deploy Script Example

```typescript
// scripts/deploy.ts
import { ethers } from 'hardhat';

async function main() {
  console.log('Deploying ConfidPay...');

  const ConfidPay = await ethers.getContractFactory('ConfidPay');
  const token = await ethers.deployContract('MockFHERC20');
  
  const contract = await ConfidPay.deploy(await token.getAddress());
  await contract.waitForDeployment();

  console.log(`ConfidPay deployed to: ${await contract.getAddress()}`);
}

main().catch((error) => {
  console.error(error);
  process.exit(1);
});
```

### Test Example

```typescript
// test/ConfidPay.test.ts
import { expect } from 'chai';
import { loadFixture } from '@nomicfoundation/hardhat-network-helpers';
import { ethers } from 'hardhat';

describe('ConfidPay', function () {
  async function deployFixture() {
    const [owner, employee] = await ethers.getSigners();

    const Token = await ethers.getContractFactory('MockFHERC20');
    const token = await Token.deploy();
    
    const ConfidPay = await ethers.getContractFactory('ConfidPay');
    const confipay = await ConfidPay.deploy(await token.getAddress());

    return { confipay, token, owner, employee };
  }

  describe('Payroll Creation', function () {
    it('should create encrypted payroll', async function () {
      const { confipay, employee } = await loadFixture(deployFixture);

      // Mock FHE encryption for testing
      const encryptedAmount = 5000000n;
      const encryptedNextRelease = BigInt(Math.floor(Date.now() / 1000) + 30 * 24 * 60 * 60);
      const encryptedVestingRate = 100n;

      await confipay.createPayroll(
        employee.address,
        encryptedAmount,
        encryptedNextRelease,
        encryptedVestingRate
      );

      const payroll = await confipay.payrolls(employee.address);
      expect(payroll.isActive).to.be.true;
    });
  });
});
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Error: Cannot find module '@cofhe/hardhat-plugin'">
    Run `npm install @cofhe/hardhat-plugin @cofhe/sdk`
  </Accordion>
  <Accordion title="Error: Invalid private key">
    Ensure your .env file has the correct format: `PRIVATE_KEY=0x...`
  </Accordion>
  <Accordion title="Error: Insufficient funds">
    Get testnet ETH from [Arbitrum Sepolia Faucet](https://www.alchemy.com/faucets/arbitrum-sepolia)
  </Accordion>
  <Accordion title="Stack too deep">
    Enable `viaIR: true` in hardhat.config.ts solidity settings
  </Accordion>
  <Accordion title="Encryption/Decryption fails">
    Ensure ACL is set correctly with `FHE.allow()`, `FHE.allowThis()`
  </Accordion>
</AccordionGroup>

## Key Patterns

### Correct ACL Setup

```solidity
function _setupACL(Payroll storage p, address recipient) internal {
    // Contract access
    FHE.allowThis(p.encryptedAmount);
    
    // Employee access (NOT allowSender!)
    FHE.allow(p.encryptedAmount, recipient);
}
```

### Correct Type Conversion

```solidity
function createPayroll(
    address recipient,
    InEuint128.Constructor calldata encryptedAmount,
    InEuint128.Constructor calldata encryptedNextRelease,
    InEuint128.Constructor calldata encryptedVestingRate
) external onlyOwner {
    Payroll storage p = payrolls[recipient];
    
    p.encryptedAmount = FHE.asEuint128(encryptedAmount);
    p.encryptedNextRelease = FHE.asEuint128(encryptedNextRelease);
    p.encryptedVestingRate = FHE.asEuint128(encryptedVestingRate);
    
    _setupACL(p, recipient);
}
```

### Correct Encrypted Conditional

```solidity
// Use FHE.select(), NOT if()
euint128 releaseAmount = FHE.select(
    FHE.le(p.encryptedNextRelease, currentTime),
    FHE.div(FHE.mul(p.encryptedAmount, p.encryptedVestingRate), FHE.asEuint128(100)),
    FHE.asEuint128(0)
);
```

# Payroll Contract

Complete ConfidPay contract with FHE encrypted payroll storage, time-based releases, vesting, milestone gating, and escrow integration.

## Key Patterns

<Warning>
**Important ACL rule:** Use `FHE.allow(value, employee)`, NOT `FHE.allowSender()`. The `allowSender` function grants access to the transaction caller (admin), not the employee.
</Warning>

<Warning>
**Type conversion:** Use `InEuint128.Constructor` for input parameters and `FHE.asEuint128()` to convert to `euint128`.
</Warning>

## Complete Contract

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@fhenixprotocol/cofhe-contracts/FHE.sol";
import "@fhenixprotocol/cofhe-contracts/operations/operations.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

interface IFHERC20 {
    function confidentialTransfer(
        address to,
        InEuint64 memory encryptedAmount
    ) external returns (euint64);
    function confidentialBalanceOf(address account) external view returns (euint64);
}

interface IConfidentialEscrow {
    function create(
        address owner,
        uint256 amount,
        address resolver,
        bytes calldata resolverData
    ) external returns (uint256);
    function fund(uint256 escrowId, uint256 amount) external;
    function redeem(uint256 escrowId) external;
    function exists(uint256 escrowId) external view returns (bool);
    function isFunded(uint256 escrowId) external view returns (bool);
}

contract ConfidPay is Ownable {
    struct Payroll {
        euint128 encryptedAmount;
        euint128 encryptedNextRelease;
        euint128 encryptedVestingRate;
        bool isActive;
    }

    enum VestingType { Immediate, Linear, Cliff }

    mapping(address => Payroll) public payrolls;
    mapping(address => euint128) public encryptedBalances;
    mapping(address => uint256[]) public employeeEscrows;
    
    IFHERC20 public confidentialToken;
    IConfidentialEscrow public escrowContract;

    uint256 public constant RELEASE_INTERVAL = 30 days;

    event PayrollCreated(address indexed recipient);
    event PaymentReleased(address indexed recipient, uint256 escrowId);

    function createPayroll(
        address recipient,
        InEuint128.Constructor calldata encryptedAmount,
        InEuint128.Constructor calldata encryptedNextRelease,
        InEuint128.Constructor calldata encryptedVestingRate
    ) external onlyOwner {
        require(!payrolls[recipient].isActive, "Payroll already exists");

        Payroll storage p = payrolls[recipient];

        p.encryptedAmount = FHE.asEuint128(encryptedAmount);
        p.encryptedNextRelease = FHE.asEuint128(encryptedNextRelease);
        p.encryptedVestingRate = FHE.asEuint128(encryptedVestingRate);
        p.isActive = true;

        _setupACL(p, recipient);

        emit PayrollCreated(recipient);
    }

    function _setupACL(Payroll storage p, address recipient) internal {
        FHE.allowThis(p.encryptedAmount);
        FHE.allow(p.encryptedAmount, recipient);

        FHE.allowThis(p.encryptedNextRelease);
        FHE.allow(p.encryptedNextRelease, recipient);

        FHE.allowThis(p.encryptedVestingRate);
        FHE.allow(p.encryptedVestingRate, recipient);
    }

    function claimPayment(address recipient) external returns (uint256 escrowId) {
        Payroll storage p = payrolls[recipient];
        require(p.isActive, "Payroll inactive");

        euint64 currentTime = FHE.asEuint64(block.timestamp);
        ebool canRelease = FHE.le(p.encryptedNextRelease, currentTime);

        euint128 releaseAmount = FHE.select(
            canRelease,
            FHE.div(
                FHE.mul(p.encryptedAmount, p.encryptedVestingRate),
                FHE.asEuint128(100)
            ),
            FHE.asEuint128(0)
        );

        if (address(escrowContract) != address(0)) {
            escrowId = escrowContract.create(
                recipient,
                0,
                address(0),
                ""
            );
            escrowContract.fund(escrowId, 0);
        }

        _updateNextRelease(p, canRelease);
        
        emit PaymentReleased(recipient, escrowId);
    }

    function _updateNextRelease(
        Payroll storage p,
        ebool wasReleased
    ) internal {
        euint128 newReleaseTime = FHE.add(
            p.encryptedNextRelease,
            FHE.asEuint128(RELEASE_INTERVAL)
        );
        p.encryptedNextRelease = FHE.select(wasReleased, newReleaseTime, p.encryptedNextRelease);
    }

    function cancelPayroll(address recipient) external onlyOwner {
        payrolls[recipient].isActive = false;
    }
}
```

## Function Reference

### createPayroll

Creates an encrypted payroll for a recipient.

```solidity
function createPayroll(
    address recipient,
    InEuint128.Constructor calldata encryptedAmount,
    InEuint128.Constructor calldata encryptedNextRelease,
    InEuint128.Constructor calldata encryptedVestingRate
) external onlyOwner
```

**Parameters:**

| Name | Type | Description |
|------|------|-------------|
| `recipient` | `address` | Recipient wallet address |
| `encryptedAmount` | `InEuint128.Constructor` | Encrypted monthly salary |
| `encryptedNextRelease` | `InEuint128.Constructor` | Encrypted first release timestamp |
| `encryptedVestingRate` | `InEuint128.Constructor` | Encrypted vesting percentage (0-100) |

**Example (SDK):**

```typescript
import { FhenixClient, InEuint128 } from '@cofhe/sdk';

const client = new FhenixClient();
const wallet = await client.encrypt(5000000); // 5 USDC

await write('createPayroll', [
  recipientAddress,
  { publicAmount: wallet },
  { publicAmount: await client.encrypt(nextRelease) },
  { publicAmount: await client.encrypt(100) },
]);
```

### claimPayment

Creates escrow for employee payment if vesting conditions are met.

```solidity
function claimPayment(address recipient) external returns (uint256 escrowId)
```

**Conditions checked (encrypted):**
- Current time >= next release timestamp (using `FHE.le()`)
- Recipient has active payroll

### cancelPayroll

Deactivates payroll (cannot be reversed).

```solidity
function cancelPayroll(address recipient) external onlyOwner
```

## ACL Setup (Correct Pattern)

```solidity
function _setupACL(Payroll storage p, address recipient) internal {
    FHE.allowThis(p.encryptedAmount);
    FHE.allow(p.encryptedAmount, recipient);  // NOT allowSender!

    FHE.allowThis(p.encryptedNextRelease);
    FHE.allow(p.encryptedNextRelease, recipient);

    FHE.allowThis(p.encryptedVestingRate);
    FHE.allow(p.encryptedVestingRate, recipient);
}
```

<Warning>
**Critical:** `FHE.allowSender()` grants access to the **transaction caller**, NOT the employee. Always use `FHE.allow(value, employee)` when creating payroll for another address.
</Warning>

## Encrypted Conditionals

<Warning>
**Never use `if (FHE.eq(...))` or `if (FHE.lt(...))` inside conditions.** Use `FHE.select()` instead:

```solidity
// ❌ WRONG - This will fail during compilation
if (FHE.eq(encryptedValue, FHE.asEuint128(0))) {
    revert("Zero value");
}

// ✅ CORRECT - Use FHE.select() for encrypted conditionals
euint128 result = FHE.select(
    FHE.eq(encryptedValue, FHE.asEuint128(0)),
    FHE.asEuint128(0),
    encryptedValue
);
```
</Warning>
